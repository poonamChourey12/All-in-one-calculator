function solveInequality() { const inequalityExpression = document.getElementById('inequalityExpression').value; try { const result = solveLinearInequality(inequalityExpression); const resultElement = document.getElementById('result'); resultElement.innerHTML = `Solution Set: ${result}`; } catch (error) { alert('Invalid inequality expression. Please enter a valid expression.'); } } function solveLinearInequality(inequalityExpression) { // Simple example: assumes the expression is in the form "ax + b < c" const match = inequalityExpression.match(/([+\-]?\d*)\s*x\s*([<>])\s*([+\-]?\d+)/); if (!match) { throw new Error('Invalid linear inequality.'); } const coefficient = match[1] ? parseInt(match[1]) : 1; const operator = match[2]; const constant = parseInt(match[3]); let solutionSet; switch (operator) { case '<': solutionSet = `x < ${(constant / coefficient).toFixed(2)}`; break; case '>': solutionSet = `x > ${(constant / coefficient).toFixed(2)}`; break; default: throw new Error('Invalid operator.'); } return solutionSet; }
